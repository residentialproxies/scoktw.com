SELECT

    AVG(digg_count) as avg_likes,
    COUNT(*) as post_count
FROM tiktok_posts
WHERE create_time >= DATE_SUB(NOW(), INTERVAL 30 DAY)
GROUP BY hashtag
HAVING post_count >= 10
ORDER BY avg_likes DESC
LIMIT 10;
